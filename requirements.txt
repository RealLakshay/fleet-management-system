# PostgreSQL via Docker (recommended)
# 1) Pull image (skip if already available locally):
#    docker pull postgres:16
#
# 2) Run container:
#    docker run -d --name fleet-postgres -e POSTGRES_DB=fleet_db -e POSTGRES_USER=fleet_user -e POSTGRES_PASSWORD=fleet_pass -p 5432:5432 postgres:16
#
# 3) Import schema:
#    docker exec -i fleet-postgres psql -U fleet_user -d fleet_db < src/main/resources/db/migration/V1__init_schema.sql
#
# 4) Update DB settings in src/main/resources/application.yml:
#    spring.datasource.url=jdbc:postgresql://localhost:5432/fleet_db
#    spring.datasource.username=fleet_user
#    spring.datasource.password=fleet_pass
#
# 5) Run from project root:
#    gradle clean build
#    gradle bootRun

# Optional: stop/remove DB container when done
#    docker stop fleet-postgres
#    docker rm fleet-postgres
